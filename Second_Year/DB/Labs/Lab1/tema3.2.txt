TRY IT/SOLVE IT:

1. SELECT employee_id AS "Number", first_name, last_name
   FROM employees
   ORDER BY "Number";

2. SELECT title
   FROM employees
   ORDER BY year, title;

3. SELECT title as "Our Collection"
   FROM d_cds
   ORDER BY title DESC;

4. SELECT first_name, last_name, student_id, parking_number
   FROM students
   WHERE year = 1
   ORDER BY last_name, first_name DESC;

5. SELECT department_id, last_name, manager_id
   FROM employees
   WHERE employee_id < 125
   ORDER BY department_id DESC, last_name DESC;
